        self.salary = salary
        self.role = role

    def print_details(self):

        print(f"""
        NAME     :{self.name}
        SALARY   :{self.salary}
        ROLE     :{self.role}
        """)

    @classmethod
    def change_leaves(cls,new_leaves):
        cls.number_of_leaves = new_leaves

    @classmethod
    def instance_with_dash(cls,params_dash):
        return cls(*params_dash.split("-"))

    @staticmethod
    def print_good_string(string):
        print(f"this is a good string : {string}")

class Subscriber:

    def __init__(self,fname,lname):
        self.fname = fname
        self.lname = lname

    def name(self):
        return f"{self.fname} {self.lname}"

    @property
    def email(self):
        if self.fname is None or self.lname is None:
            return "Email is not set"
        return f"{self.fname}.{self.lname}@faloola.com"

    @email.setter
    def email(self,new_email):
        name = new_email.split("@")[0]
        name_l = name.split(".")
        self.fname =  name_l[0]
        self.lname = name_l[1]

    @email.deleter
    def email(self):
        self.fname = None
        self.lname = None
